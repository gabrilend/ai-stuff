#!/bin/bash
set -euo pipefail

FILE_DIR="/home/ritz/words"

if [ -z "${FILE_DIR}" ]; then
   echo "please enter the location of the words directory at the top of the 
script."
   exit 1
fi


function view(){
   DIR="${FILE_DIR}/${1}"

   foreground_color=0
   background_color=4
   while [ "${background_color}" -eq 4 ]; do
      background_color=$(shuf -i 0-7 -n 1)
   done

   if [ "${background_color}" -eq 0 ]; then
      foreground_color=$(shuf -i 1-7 -n 1)
      while [ "${foreground_color}" -eq 4 ]; do
         foreground_color=$(shuf -i 1-7 -n 1)
      done
   fi


   echo -e -n "\033[3${foreground_color}m"
   echo -e -n "\033[4${background_color}m"
   FILE="$(ls -1 ${DIR} | shuf -n 1)"
   while test -d "${DIR}/${FILE}"; do
      DIR="${DIR}/${FILE}"
      FILE="$(ls -1 ${DIR} | shuf -n 1)"
   done
   cat "${DIR}/${FILE}" | fold -w80 -s
   echo -e -n "\033[39m"
   echo -e "\033[49m"
   echo ""
   echo "file: ${DIR}/${FILE}" | fold -w80
}

function view-random(){
   function_count=3
   functions=("fediverse" "messages" "notes")
   random_index=$((RANDOM%$function_count))
   view "${functions[$random_index]}"
}
view-random
