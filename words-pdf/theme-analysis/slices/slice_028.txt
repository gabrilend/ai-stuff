    if [ "${1}" = "..." ] ; then

        builtin cd .. && builtin cd .. && builtin cd ..

    elif [ "${1}" = "cdir" ] ; then
        local target_dir="$(tail -n 1 '/home/ritz/scripts/.cdir-target')"
        echo ${target_dir}

        local target_dir="${1}"
    fi
